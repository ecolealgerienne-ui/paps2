export function useUpdateProductConfig(farmId: string) {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: ({ configId, data }: { configId: string; data: UpdateConfigDto }) =>
      preferencesApi.updateConfig(farmId, configId, data),
    onSuccess: () => {
      queryClient.invalidateQueries(['farm-product-preferences', farmId]);
    },
  });
}

// hooks/useFarmerLots.ts

export function useFarmerLots(farmId: string, configId: string) {
  return useQuery({
    queryKey: ['farmer-lots', farmId, configId],
    queryFn: () => lotsApi.list(farmId, configId),
  });
}

export function useCreateLot(farmId: string, configId: string) {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: (data: CreateLotDto) =>
      lotsApi.create(farmId, configId, data),
    onSuccess: () => {
      queryClient.invalidateQueries(['farmer-lots', farmId, configId]);
      queryClient.invalidateQueries(['farm-product-preferences', farmId]);
    },
  });
}

// utils/errorHandler.ts

export function handleApiError(error: unknown): string {
  if (error instanceof ApiError) {
    // Erreur API connue
    return i18n.t(error.messageKey) || error.message;
  }

  if (error instanceof ZodError) {
    // Erreur validation
    return error.errors.map(e => e.message).join(', ');
  }

  // Erreur inconnue
  return i18n.t('errors.generic');
}

// Usage dans composant

  toast.error(handleApiError(error));
}
